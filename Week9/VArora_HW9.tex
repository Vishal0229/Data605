\[g(t)=\sum _{ j=0 }^{ n }{ { e }^{ tj } } (n!/(n-j)!j!){ p }^{ j }{ q }^{ n-j }\]
\[g(t)=\sum _{ j=0 }^{ n }{ (n!/(n-j)!j!) } { { (pe }^{ t }) }^{ j }{ q }^{ n-j }\]
\[g(t)={ ({ pe }^{ t }+q) }^{ n }\]
\[g'(t)=n{ ({ pe }^{ t }+q) }^{ n-1 }{ pe }^{ t }\]
